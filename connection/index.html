// 呼叫後端 API 來產生短網址 const response = await fetch('/api/generate-short-url', { method: 'POST', headers: { 'Content-Type': 'application/json', }, body: JSON.stringify({ longUrl: longUrl }), }); if (!response.ok) { const errorData = await response.json(); throw new Error(errorData.error || '產生短網址失敗'); } const data = await response.json(); const shortUrl = data.shortUrl; // 從後端取得短網址 // 清空之前的結果和錯誤 errorDiv.textContent = ''; qrCodeContainer.innerHTML = ''; resultDiv.style.display = 'block'; try { // --- 生成帶有 Logo 的 QR Code --- // 1. 建立一個隱藏的 div 來生成 QR Code const hiddenDiv = document.createElement('div'); hiddenDiv.style.display = 'none'; document.body.appendChild(hiddenDiv); // 2. 使用 qrcode.js 生成 QR Code 到隱藏的 div const qrCode = new QRCode(hiddenDiv, { text: shortUrl, width: 256, height: 256, colorDark: "#000000", colorLight: "#ffffff", correctLevel: QRCode.CorrectLevel.H }); // 3. 監聽 img 元素的 onload 事件 const img = hiddenDiv.querySelector('img'); if (img) { img.onload = function() { // 4. 設定 canvas 大小 canvas.width = 256; canvas.height = 256; // 5. 將 QR Code 圖片畫到 canvas 上 ctx.drawImage(img, 0, 0); // 6. 載入 Logo 圖片 const logoImg = new Image(); logoImg.src = 'logo.png'; // 假設 logo.png 放在 connection 資料夾下 logoImg.onload = function() { // 7. 計算 Logo 的大小和位置 const logoSize = Math.min(256, 256) * 0.2; const x = (256 - logoSize) / 2; const y = (256 - logoSize) / 2; // 8. 將 Logo 畫到 canvas 上 ctx.drawImage(logoImg, x, y, logoSize, logoSize); // 9. 顯示結果區域 resultDiv.style.display = 'block'; // 10. 移除隱藏的 div document.body.removeChild(hiddenDiv); }; logoImg.onerror = function() { console.error('Logo 圖片載入失敗！'); // 如果 Logo 載入失敗，仍然顯示 QR Code resultDiv.style.display = 'block'; document.body.removeChild(hiddenDiv); }; }; img.onerror = function() { console.error('QR Code 圖片生成失敗！'); showError('QR Code 生成失敗'); document.body.removeChild(hiddenDiv); }; } else { console.error('QR Code 圖片生成失敗！'); showError('QR Code 生成失敗'); document.body.removeChild(hiddenDiv); } } catch (err) { console.error('Error:', err); showError(err.message || '發生錯誤'); }
